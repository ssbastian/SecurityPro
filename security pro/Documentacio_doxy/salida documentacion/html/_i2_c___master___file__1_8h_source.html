37  RCEN = 1; /*enable receive*/
38  while(!SSPSTATbits.BF); /*wait for buffer full flag

50 void I2C_Init()
51 {
52  TRISB0=1; /*set up I2C lines by setting as input*/
53  TRISB1=1;
54  SSPSTAT=80; /*slew rate disabled, other bits are cleared*/
55  SSPCON1=0x28; /*enable SSP port for I2C Master mode,
56  *clock = FOSC / (4 * (SSPADD+1))*/
57  SSPCON2=0;
58  SSPADD=BITRATE; /*clock 100 kHz*/
59  SSPIE=1; /*enable SSPIF interrupt*/
60  SSPIF=0;
61 }

12 char I2C_Read(char flag) /*read data from location and
13  *send ack or nack depending upon parameter*/
14 {
15  char buffer;
16  RCEN = 1; /*enable receive*/
17  while(!SSPSTATbits.BF); /*wait for buffer full flag
18  *which is set when complete byte received*/
19  buffer = SSPBUF; /*copy SSPBUF to buffer*/
20  /*send acknowledgment or negative acknowledgment */
21  if(flag==0)
22  I2C_Ack();

24  I2C_Nack();
25  I2C_Ready();

32  TRISB1=1;
33  SSPSTAT=80; /*slew rate disabled, other bits are cleared*/
